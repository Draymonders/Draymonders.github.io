#include <bits/stdc++.h>
using namespace std;

void quickSort(vector<int>& nums, int l, int r) {
    if (l >= r) return ;
    int i = l, j = r;
    int val = nums[l];
    while (i < j) {
        while (i < j && nums[j] >= val) j--;
        while (i < j && nums[i] <= val) i++;
        if (i < j) {
            swap(nums[i], nums[j]);
        }
    }
    swap(nums[l], nums[i]);
    quickSort(nums, l, i);
    quickSort(nums, i+1, r);
    return ;
}

int main()
{
    vector<int> v = {1,3,2,1,4,5};
    quickSort(v, 0, v.size()-1);
    for (auto x : v) {
        cout << x << " ";
    }
    cout << endl;
    return 0;
}
