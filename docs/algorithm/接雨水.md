# 接雨水

使用单调递减的栈，当栈顶的值 < 当前元素的高度，说明 栈顶处于山谷(`h[st.top().top()] > h[st.top()] < h[i]`)

因此中间会有雨水

```cpp
class Solution {
public:
    int trap(vector<int>& h) {
        // 单调递减的栈
        stack<int> st;
        int n = h.size();
        int res = 0;

        for (int i=0; i<n; i++) {
            while (!st.empty() && h[st.top()] < h[i]) {
                int f = st.top(); st.pop();
                if (!st.empty()) {
                    // printf("[%d, %d] h: %d\n", st.top(), i, min(h[i], h[st.top()]));
                    res += (i - st.top() - 1) * (min(h[i], h[st.top()]) - h[f]);
                    // printf("res: %d\n", res);
                } 
            }
            st.push(i);
        }
        return res;
    }
};
```
